int weightedSum(string s) {
vector<int> levelSum(1000, 0); // depth -> sum of ints at this depth
int depth = 0;
int maxDepth = 0;



int num = 0;
bool reading = false;

for (char c : s) {
    if (c == '[') {
        depth++;
        maxDepth = max(maxDepth, depth);
    } 
    else if (isdigit(c)) {
        reading = true;
        num = num * 10 + (c - '0');
    }
    else if (c == ',' || c == ']') {
        if (reading) {
            levelSum[depth] += num;
            num = 0;
            reading = false;
        }
        if (c == ']') depth--;
    }
}

int total = 0;
for (int d = 1; d <= maxDepth; d++)
    total += levelSum[d] * (maxDepth - d + 1);

return total;




}
